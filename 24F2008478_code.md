```python
import os
import cv2
import torch
import torch.nn as nn
import torch.optim as optim
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from torch.utils.data import Dataset, DataLoader
from torchvision import models, transforms
from PIL import Image
from tqdm.auto import tqdm
from sklearn.model_selection import train_test_split

# 1. Environment Setup
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
BASE_PATH = '/kaggle/input/competitions/26-t-1-dl-gen-ainppe-1'
IMAGE_DIR = os.path.join(BASE_PATH, 'images')

# 2. Step 1: Understand the Problem & Structure
train_df = pd.read_csv(os.path.join(BASE_PATH, 'train.csv'))
test_df = pd.read_csv(os.path.join(BASE_PATH, 'test.csv'))
target_cols = [col for col in train_df.columns if col != 'id']

# 3. Step 2: EDA (Exploratory Data Analysis)
class_counts = train_df[target_cols].sum().sort_values(ascending=False)
plt.figure(figsize=(12, 6))
sns.barplot(x=class_counts.values, y=class_counts.index, palette='magma')
plt.title("Class Distribution - 24F2008478")
plt.show()

# 4. Step 3: Data Preprocessing & Dataset Class
train_data, val_data = train_test_split(train_df, test_size=0.15, random_state=42)

class ChestDataset(Dataset):
    def __init__(self, df, img_dir, transform=None, is_test=False):
        self.df = df
        self.img_dir = img_dir
        self.transform = transform
        self.is_test = is_test

    def __len__(self):
        return len(self.df)

    def __getitem__(self, idx):
        img_name = self.df.iloc[idx]['id']
        img_path = os.path.join(self.img_dir, img_name)
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        
        # CLAHE (Medical Image Enhancement)
        lab = cv2.cvtColor(image, cv2.COLOR_RGB2LAB)
        l, a, b = cv2.split(lab)
        clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8,8))
        l = clahe.apply(l)
        image = cv2.merge((l,a,b))
        image = cv2.cvtColor(image, cv2.COLOR_LAB2RGB)
        
        image = Image.fromarray(image)
        if self.transform: image = self.transform(image)
        if self.is_test: return image, img_name
        
        labels = torch.tensor(self.df.iloc[idx][target_cols].values.astype(float), dtype=torch.float32)
        return image, labels

data_transforms = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.RandomHorizontalFlip(),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225])
])

train_loader = DataLoader(ChestDataset(train_data, IMAGE_DIR, data_transforms), batch_size=32, shuffle=True)
val_loader = DataLoader(ChestDataset(val_data, IMAGE_DIR, data_transforms), batch_size=32, shuffle=False)

# 5. Step 4: Model Development (DenseNet + Combined Loss)
model = models.densenet121(weights='IMAGENET1K_V1')
model.classifier = nn.Linear(model.classifier.in_features, 20)
model = model.to(device)

class CombinedAsymmetricLoss(nn.Module):
    def __init__(self, pos_weight, gamma=2.0):
        super(CombinedAsymmetricLoss, self).__init__()
        self.bce = nn.BCEWithLogitsLoss(pos_weight=pos_weight)
        self.gamma = gamma

    def forward(self, inputs, targets):
        bce_loss = self.bce(inputs, targets)
        probs = torch.sigmoid(inputs)
        pt = targets * probs + (1 - targets) * (1 - probs)
        focal_weight = (1 - pt) ** self.gamma
        return (bce_loss * focal_weight).mean()

# Asymmetric Logic: 5x weight for pathologies to minimize -5 penalty
pos_weights = torch.ones(20).to(device)
pos_weights[0:19] = 5.0 
pos_weights[19] = 1.0

criterion = CombinedAsymmetricLoss(pos_weight=pos_weights, gamma=2.0)
optimizer = optim.Adam(model.parameters(), lr=1e-4)
scheduler = optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.1, patience=2)

# 6. Step 5: Model Training & Monitoring
for epoch in range(4):
    model.train()
    running_loss = 0.0
    for images, labels in tqdm(train_loader, desc=f"Epoch {epoch+1} [Train]"):
        images, labels = images.to(device), labels.to(device)
        optimizer.zero_grad()
        loss = criterion(model(images), labels)
        loss.backward()
        optimizer.step()
        running_loss += loss.item()
    
    model.eval()
    val_loss = 0.0
    with torch.no_grad():
        for images, labels in val_loader:
            images, labels = images.to(device), labels.to(device)
            val_loss += criterion(model(images), labels).item()
    
    scheduler.step(val_loss/len(val_loader))
    print(f"Epoch {epoch+1} Complete. Val Loss: {val_loss/len(val_loader):.4f}")

# 7. Step 6 & 7: Inference & Submission Format
test_loader = DataLoader(ChestDataset(test_df, IMAGE_DIR, transform=data_transforms, is_test=True), batch_size=32, shuffle=False)
model.eval()
all_probs, all_ids = [], []

with torch.no_grad():
    for images, ids in tqdm(test_loader, desc="Predicting"):
        images = images.to(device)
        probs = torch.sigmoid(model(images))
        all_probs.append(probs.cpu().numpy())
        all_ids.extend(ids)

predictions = np.vstack(all_probs)
predicted_indices = np.argmax(predictions, axis=1)

sub_df = pd.DataFrame(all_ids, columns=['id'])
submission_matrix = np.zeros_like(predictions)
for i, idx in enumerate(predicted_indices):
    submission_matrix[i, idx] = 1

for i, col in enumerate(target_cols):
    sub_df[col] = submission_matrix[:, i].astype(int)

# Final Verification and Save
if all(sub_df[target_cols].sum(axis=1) == 1):
    sub_df.to_csv('submission.csv', index=False)
    print("✅ submission.csv saved successfully!")
