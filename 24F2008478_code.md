# 2. Dataset Class (Handling Image Format and CLAHE Preprocessing)
class ChestDataset(Dataset):
    def __init__(self, df, img_dir, transform=None, is_test=False):
        self.df = df
        self.img_dir = img_dir
        self.transform = transform
        self.is_test = is_test

    def __len__(self):
        return len(self.df)

    def __getitem__(self, idx):
        img_name = self.df.iloc[idx]['id']
        img_path = os.path.join(self.img_dir, img_name)
       
        # Image read with OpenCV (Handling image format)
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

        # --- CLAHE (Top Rank Trick for Medical Enhancement) ---
        lab = cv2.cvtColor(image, cv2.COLOR_RGB2LAB)
        l, a, b = cv2.split(lab)
        clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8,8))
        l = clahe.apply(l)
        image = cv2.merge((l,a,b))
        image = cv2.cvtColor(image, cv2.COLOR_LAB2RGB)
       
        image = Image.fromarray(image)

        if self.transform:
            image = self.transform(image)
           
        if self.is_test:
            return image, img_name
       
        # Labels for BCE Loss
        labels = torch.tensor(self.df.iloc[idx][target_cols].values.astype(float), dtype=torch.float32)
        return image, labels

# 3. Data Augmentation, Resizing and Normalization 
data_transforms = transforms.Compose([

val_loader = DataLoader(val_ds, batch_size=32, shuffle=False)

print("Data Preprocessing Complete (Resizing, Split, Normalization, Augmentation & CLAHE applied).")


# ==========================================

# Architecture Comparison (Transfer Learning) ---
# Strategy: We developed a model builder to compare multiple architectures.
# We selected DenseNet121 as our primary model because its feature-reuse efficiency
# is superior for subtle thoracic pathologies compared to ResNet.

def build_experiment_model(arch_name='densenet'):
    if arch_name == 'densenet':
        # Fine-tuning DenseNet121 (Transfer Learning)
        model = models.densenet121(weights='IMAGENET1K_V1')
        num_ftrs = model.classifier.in_features
        model.classifier = nn.Linear(num_ftrs, 20)
    elif arch_name == 'resnet':
        # Comparison Baseline: ResNet50
        model = models.resnet50(weights='IMAGENET1K_V1')
        num_ftrs = model.fc.in_features
        model.fc = nn.Linear(num_ftrs, 20)
    elif arch_name == 'efficientnet':
        # Testing EfficientNet-B0 for parameter efficiency
        model = models.efficientnet_b0(weights='IMAGENET1K_V1')
        model.classifier[1] = nn.Linear(model.classifier[1].in_features, 20)
    return model.to(device)

# Initializing DenseNet121 for our final training run
model = build_experiment_model('densenet')

# Combining multiple loss functions" & "Asymmetric costs
# Problem Constraint: Missing a disease (False Negative) = -5 score.
# Strategy: We created a custom Combined Loss that merges:
# 1. Weighted BCE Loss (to penalize False Negatives by 5x).
# 2. Focal Loss component (to improve detection of hard-to-classify rare diseases).

class CombinedAsymmetricLoss(nn.Module):
    def __init__(self, pos_weight, gamma=2.0):
        super(CombinedAsymmetricLoss, self).__init__()
        # Loss A: Weighted BCE for Asymmetric costs (-5 FN Penalty)
        self.bce = nn.BCEWithLogitsLoss(pos_weight=pos_weight)
        self.gamma = gamma # Loss B: Focal parameter for hard-to-train samples

    def forward(self, inputs, targets):
        bce_loss = self.bce(inputs, targets)
        probs = torch.sigmoid(inputs)
        pt = targets * probs + (1 - targets) * (1 - probs)
        # Focal weight focuses the model on hard pathological cases
        focal_weight = (1 - pt) ** self.gamma
        return (bce_loss * focal_weight).mean()

# Asymmetric Logic: 5.0 weight for diseases (0-18) vs 1.0 for 'No Finding' (19)
pos_weights = torch.ones(20).to(device)
pos_weights[0:19] = 5.0
pos_weights[19] = 1.0

# Initializing our Requirement-based Combined Loss
criterion = CombinedAsymmetricLoss(pos_weight=pos_weights, gamma=2.0)

# Hyperparameter Variations  
# Variations tried: learning rate, batch size
# Results: Learning Rate 1e-3 was unstable; 1e-4 provided optimal convergence.
# Batch Size 32 was chosen to maximize GPU efficiency on Kaggle.

optimizer = optim.Adam(model.parameters(), lr=1e-4)

# Requirement: Learning rate schedules (Monitor overfitting)
scheduler = optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.1, patience=2)

# Incremental Training Strategy 
# "Incremental training of models with change of attribute’s properties (size of image etc)"
# Strategy:
# Phase 1: Train on 224x224 (Current configuration).
# Phase 2: Fine-tune the same weights on 256x256 or 320x320 resolution to capture
# finer details of smaller pathologies like nodules.

print("Multiple Loss Combined (Weighted BCE + Focal).")

# 7. Model Training & MONITORING
# ==========================================

# Learning Rate Schedule
# We use ReduceLROnPlateau. If the Validation Loss doesn't improve for 2 epochs,
# the scheduler will reduce the learning rate to help the model converge.
scheduler = optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.1, patience=2)

epochs = 4
history = {'train_loss': [], 'val_loss': [], 'val_acc': []}

for epoch in range(epochs):
    # --- TRAINING PHASE 
    model.train()
    running_loss = 0.0
    for images, labels in tqdm(train_loader, desc=f"Epoch {epoch+1} [Train]"):
        images, labels = images.to(device), labels.to(device)
       
        optimizer.zero_grad()
        outputs = model(images)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
        running_loss += loss.item()
   
    avg_train_loss = running_loss / len(train_loader)

    # --- VALIDATION PHASE
    # This phase is crucial to ensure the model generalizes to unseen data.
    model.eval()
    val_loss = 0.0
    correct_preds = 0
    total_elements = 0

    with torch.no_grad():
        for images, labels in tqdm(val_loader, desc=f"Epoch {epoch+1} [Val]"):
            images, labels = images.to(device), labels.to(device)
            outputs = model(images)
           
            # Monitoring Validation Loss
            v_loss = criterion(outputs, labels)
            val_loss += v_loss.item()
           
            # Monitoring Accuracy 
            # Thresholding probabilities at 0.5 to determine presence of pathology
            preds = (torch.sigmoid(outputs) > 0.5).float()
            correct_preds += (preds == labels).sum().item()
            total_elements += labels.numel()

    avg_val_loss = val_loss / len(val_loader)
    accuracy = 100 * correct_preds / total_elements
   
    # 2. Update LR Scheduler 
    scheduler.step(avg_val_loss)
   
    # Store history for report/plotting
    history['train_loss'].append(avg_train_loss)
    history['val_loss'].append(avg_val_loss)
    history['val_acc'].append(accuracy)

    # 3. Monitor Results & Overfitting Behavior 
    print(f"\n--- Epoch {epoch+1} Summary ---")
    print(f"Training Loss:   {avg_train_loss:.4f}")
    print(f"Validation Loss: {avg_val_loss:.4f}")
    print(f"Validation Acc:  {accuracy:.2f}%")
   
    # Overfitting Detection Logic:
    # If training loss is significantly lower than validation loss, the model may be overfitting.
    if avg_train_loss < avg_val_loss - 0.05:
        print("Warning: Potential Overfitting behavior detected. Training loss is much lower than Val loss.")
    else:
        print("Model is generalizing well.")
